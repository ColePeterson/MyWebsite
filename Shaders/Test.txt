precision mediump float;
uniform vec2 resolution;
uniform float time;
uniform vec2 mouse;

vec2 R;


void main()
{
    R = resolution.xy;
    vec2 uv = vec2(gl_FragCoord.xy - 0.5*R.xy)/R.y;
    
   
    
	vec2 m = mouse.xy / R.xy -.5; 
	m.x *= R.x/R.y;
    
    vec3 col = vec3(1);
	

	
    col *= smoothstep(0.05,0.06,length(uv-m));
	

    
    gl_FragColor = vec4(col, 1.0);
    
}
