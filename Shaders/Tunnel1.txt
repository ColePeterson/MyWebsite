precision mediump float;
uniform vec2 resolution;
uniform float time;



void main()
{
    

vec2 f = 2.0 * vec2(gl_FragCoord.xy - 0.5*resolution.xy)/resolution.y;


vec2 b = vec2(f.x / f.y, f.y / f.x);

vec2 u = ceil(vec2(1.0 / length(b) - 0.28, time*1.2 + 1.0 / length(f))*5.0);
 
vec3 t = vec3(1) * mod(u.x - u.y, 2.0);


gl_FragColor = vec4(t * smoothstep(0.15, 0.9, length(f)), 1.0);
 


}
